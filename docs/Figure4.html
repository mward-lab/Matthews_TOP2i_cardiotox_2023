colnames(FCmatrix) <-
  c(
    "DNR\n3h",

    "EPI\n3h",
    "MTX\n3h",
    "TRZ\n3h",
    "DNR\n24h",

    "EPI\n24h",
    "MTX\n24h",
    "TRZ\n24h"
  )


mat_col <-
  data.frame(
    time = c(rep("3 hours", 5), rep("24 hours", 5)),
    class = (c(
      "AC", "AC", "AC", "nAC", "nAC", "AC", "AC", "AC", "nAC", "nAC"
    )),
    TOP2i = c(rep("yes", 4), "no", rep("yes", 4), "no")
  )
rownames(mat_col) <- colnames(FCmatrix)

mat_colors <-
  list(
    time = c("pink", "chocolate4"),
    class = c("yellow1", "lightgreen"),
